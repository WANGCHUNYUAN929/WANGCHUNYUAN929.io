    vue-cdn 性能优化

    # 1.根据环境判断是否需要cdn加载, 开发环境当前不用, 打包上线才走cdn操作

    const isProd = process.env.NODE_ENV === 'production'
    1

    # 2.配置需要加载的cdn列表 cdn推荐https://www.jsdelivr.com/

    const assetsCDN = {
    // externals 忽略打包文件 key名为cdn暴露的属性名自行查看
    externals: {
        'vue-router': 'VueRouter',
    },
    js: [
        'https://cdn.jsdelivr.net/npm/vue@2.6.14/dist/vue.min.js',
    ],
    css: []

    示例代码 vue.config.js

    /*
 * @Descripttion:
 * @version:
 * @Author: WANGCHUNYUAN
 * @Date: 2021-11-19 22:13:59
 * @LastEditTime: 2021-12-23 19:47:01
 */
const path = require('path')
const webpack = require('webpack')
const isProd = process.env.NODE_ENV === 'production'
console.log(isProd, 'isProd')
const BundleAnalyzerPlugin = require('webpack-bundle-analyzer').BundleAnalyzerPlugin
const assetsCDN = {
    externals: {
        vue: 'Vue',
        'vue-router': 'VueRouter',
        vuex: 'Vuex',
        axios: 'axios',
        moment: 'moment'
    },
    js: [
        'https://cdn.jsdelivr.net/npm/vue@2.6.14/dist/vue.min.js',
        'https://cdn.jsdelivr.net/npm/vue-router@3.5.3/dist/vue-router.min.js',
        'https://cdn.jsdelivr.net/npm/vuex@3.6.2/dist/vuex.min.js',
        'https://cdn.jsdelivr.net/npm/axios@0.21.1/dist/axios.min.js',
        'https://cdn.jsdelivr.net/npm/moment@2.29.1/moment.min.js'
    ],
    // 'https://unpkg.com/element-ui@2.8.2/lib/theme-chalk/index.css'
    css: []
}
module.exports = {
    outputDir: `dist`,
    publicPath: './',
    chainWebpack: (config) => {
        config.resolve.alias.set('@js', path.join(__dirname, './src/assets/js'))
        config.resolve.alias.set('@css', path.join(__dirname, './src/assets/css'))
        config.resolve.alias.set('@img', path.join(__dirname, './src/style'))
        config.resolve.alias.set('@c', path.join(__dirname, './src/components'))
        // if prod is on assets require on cdn
        if (isProd) {
            config.plugin('html').tap((args) => {
                args[0].cdn = assetsCDN
                return args
            })
        }
    },
    css: {
        loaderOptions: {
            sass: {
                prependData: `@import "~@css/variable.scss";`
            },
            scss: {
                prependData: `@import "~@css/variable.scss";`
            }
        },
        sourceMap: true
    },
    configureWebpack: {
        externals: isProd ? assetsCDN.externals : {},
        // new BundleAnalyzerPlugin(), 打包分析
        plugins: [new webpack.IgnorePlugin(/^\.\/locale$/, /moment$/), new BundleAnalyzerPlugin()]
    }
}
